import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;

public class Program
{
    public static void main(String[] args) throws IOException {
		
                BufferedReader nameInput = new BufferedReader(new InputStreamReader(System.in));
                
                int nameLimit = 10;
                String names[] = new String[nameLimit];

                for(int x=0;x<nameLimit;x++){
                    int numList = x+1;
                    System.out.print("Enter name #"+numList+": ");
                    String name = nameInput.readLine();
                    names[x] = name;
                }
                
                System.out.println("\n\nList of Names:");
                for(int x=0;x<nameLimit;x++){
                    int numList = x+1;
                    System.out.println(numList+". "+names[x]);
                }

	}
}
