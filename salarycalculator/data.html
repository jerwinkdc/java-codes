import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.text.DecimalFormat;

//@author Jerwin

//Inheritance OOP Implementation
public class SalaryData extends SalaryMethods {
    
    final BufferedReader userInput = new BufferedReader(new InputStreamReader(System.in));
    private boolean errorOccured = false;
    public String employee;
    public int daysWorked;
    public float gross, philhealth_contribution, sss_contribution, pagibig_contribution;
            
    public void getInputs() throws IOException {
        System.out.print("Enter Name: ");
        employee = userInput.readLine();
        daysWorked = getNumberInput();
    }
    
    private int getNumberInput() throws IOException{
      
        int input = 0;
        do{
                System.out.print("Enter No. of days worked: ");
                errorOccured = false;
                try {
                    input = Integer.parseInt(userInput.readLine());
                }
                catch(NumberFormatException e){
                    errorOccured = true;
                }
            }while(errorOccured == true);
        
        return input;
    }
    
    private float getDeduction(String deduction){
        deduction = deduction.replace("%", "");
        return (Float.parseFloat(deduction) / 100);
    }
    
    public String formatOutput(float output){
        return new DecimalFormat("#,###.00").format(output);
    }

        public String get_PhilHealth_contribution(){
            philhealth_contribution = gross * getDeduction("2.75%");
            return "\u20B1"+formatOutput(philhealth_contribution);
        }
        
        public String get_SSS_contribution(){
            sss_contribution = gross * getDeduction("3.36%");
            return "\u20B1"+formatOutput(sss_contribution);
        }
        
        public String get_PagIbig_contribution(){
            pagibig_contribution = gross * getDeduction("2%");
            return "\u20B1"+formatOutput(pagibig_contribution);
        }
    
    

}
