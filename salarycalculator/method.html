public class SalaryMethods {
  private float gross_pay, total_deducations, net_pay;
  final float wage = (float)500.00;
  
        public String get_Wage(){
            return "\u20B1"+String.format("%.2f", wage);
        }
  
        public String get_GrossSalary(SalaryData data){
            gross_pay = data.daysWorked * wage;
            data.gross = gross_pay;
            return "\u20B1"+data.formatOutput(gross_pay);
        }
        
        public String get_TotalDeduction(SalaryData data){
            total_deducations = data.philhealth_contribution + 
                                data.sss_contribution +
                                data.pagibig_contribution;
            return "\u20B1"+data.formatOutput(total_deducations);
        }
        
        public String get_NetSalary(SalaryData data){
            net_pay = gross_pay - total_deducations;
            return "\u20B1"+data.formatOutput(net_pay);
        }

}
