package delacruz.lab3;

import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;

//@author Jerwin

class Calc {

    //VARIABLES
    boolean errorOccured = false;
    BufferedReader userInput = new BufferedReader(new InputStreamReader(System.in));
    int userCurrentChoice;
    double a, b, c, radius, height;
    
    //MENU
    void mainMenu() throws IOException {
            System.out.println("1. Addition");
            System.out.println("2. Subtraction");
            System.out.println("3. Area of Circle");
            System.out.println("4. Area of Cylinder");
            System.out.println("----------------------");
            System.out.print("Enter your choice: ");
            userCurrentChoice = getMenuInput();
    }
    
    //METHODS
    void getNormalInputs() throws IOException {
        System.out.println();
        System.out.print("Enter first number: ");
        a = getNumberInput();
        System.out.print("Enter second number: ");
        b = getNumberInput();
        System.out.print("Enter third number: ");
        c = getNumberInput();
        System.out.println();
    }

    void getCircleInput() throws IOException {
       System.out.println();
       System.out.print("Enter the radius: ");
       radius = getNumberInput();
       System.out.println();
    }
    
    void getCylinderInputs() throws IOException {
       System.out.println();
       System.out.print("Enter the radius: ");
       radius = getNumberInput();
       System.out.print("Enter the height: ");
       height = getNumberInput();
       System.out.println();
    }
    
    //INPUT GETTER SETTER
    int getMenuInput() throws IOException{
        int input;
              try { input = Integer.parseInt(userInput.readLine()); }
              catch(NumberFormatException e) { input = 0; }
        return input;
    }
    
    double getNumberInput() throws IOException{
        double input = 0;
        try {
            input = Double.parseDouble(userInput.readLine());
        }
        catch(NumberFormatException e){
            errorOccured = true;
        }
        return input;
    }
    
   String getNormalOutput(double input){
       int cinput = (int) input;
       if(cinput == input){ return String.valueOf(cinput); }
       else{ return String.valueOf(input); }
   }
   
   String getCircleOutput(double radius){
       double minput = Math.PI*Math.pow(radius, 2);
       double rinput = Double.parseDouble(String.format("%.2f", minput));
       int cinput = (int) minput;
       if(cinput == minput){ return String.valueOf(cinput); }
       else{ return String.valueOf(rinput); }
   }
   
   String getCylinderOutput(double radius, double height){
       double minput = (2*Math.PI*Math.pow(radius, 2))+(2*Math.PI*radius*height);
       double rinput = Double.parseDouble(String.format("%.2f", minput));
       int cinput = (int) minput;
       if(cinput == minput){ return String.valueOf(cinput); }
       else{ return String.valueOf(rinput); }
   }
    
    //MENU RETURNS
    public int choice() {
        return userCurrentChoice;
    }

    //MAIN RETURNS
    String addition() {
       String answer;
       if(errorOccured == true){ answer = "undefined."; }
       else{ answer = getNormalOutput(a+b+c)+"!"; }
       return answer;
    }

    String subtraction() {
       String answer;
       if(errorOccured == true){ answer = "undefined."; }
       else{ answer = getNormalOutput(a-b-c)+"!"; }
       return answer;
    }

    String AreaCircle() {
       String answer;
       if(errorOccured == true){ answer = "undefined."; }
       else{ answer = getCircleOutput(radius)+"!"; }
       return answer;
    }

    String AreaCylinder() {
       String answer;
       if(errorOccured == true){ answer = "undefined."; }
       else{ answer = getCylinderOutput(radius,height)+"!"; }
       return answer;
    }
    
}
