        public int calculateAge(String birthday) {
            DateTimeFormatter formatter = DateTimeFormatter.ofPattern("M/d/yyyy");
            LocalDate birthDate = LocalDate.parse(birthday, formatter);
            LocalDate currentDate = LocalDate.now();
            if ((birthDate != null) && (currentDate != null)) {
                return Period.between(birthDate, currentDate).getYears();
            } else {
                return 0;
            }
        }

    private static class StudentGrades{
        
        final BufferedReader userInputGrades = new BufferedReader(new InputStreamReader(System.in));
        private boolean errorOccured = false;
        private float ptaskG, portfolioG, iasgmtG;
         
           public String getGrade(String display, String target, int rawScore) throws IOException{
                float score = 0, grade = 0;
                do{
                        System.out.print(display);
                        errorOccured = false;
                        try {
                            score = Float.parseFloat(userInputGrades.readLine());
                        }
                        catch(NumberFormatException e){
                            errorOccured = true;
                        }
                            if(rawScore < (float) 50){
                                 System.out.println("\nRaw Score should be equal or higher than 50!");
                                 System.exit(0);
                            }
                            
                            if(score < (float) 0 || score > (float) rawScore){
                                errorOccured = true;
                            }
                            
                            if(errorOccured == false){

                                if("ptaskS".equals(target)){
                                    grade = (float) (((score/rawScore)*50+50)*0.5);
                                    ptaskG = (grade);
                                }
                                else if("portfolioS".equals(target)){
                                    grade = (float) (((score/rawScore)*50+50)*0.25);
                                    portfolioG = (grade);
                                }
                                else if("iasgmtS".equals(target)){
                                    grade = (float) (((score/rawScore)*50+50)*0.25);
                                    iasgmtG = (grade);
                                }
                                else{
                                    System.out.println("\nInvalid task detected,\nmake sure to put a valid task!");
                                    System.exit(0);
                                }

                            }
                    }while(errorOccured == true);

                return formatOutput(grade)+"%";
        }
           
        public String getFinalGrade(String passing) {
            String remarks = "(passed)";
            float fgradeTotal = 0;
            
            if(passing.contains("Passing:")){
                float passingGrade = Float.parseFloat(passing.replace("Passing:",""));
                //System.out.println("Passing: "+formatOutput(passingGrade));
                //System.out.println("PT: "+formatOutput(ptaskG));
                //System.out.println("P: "+formatOutput(portfolioG));
                //System.out.println("IA: "+formatOutput(iasgmtG));

                fgradeTotal = ptaskG + portfolioG + iasgmtG;

                if(fgradeTotal < passingGrade){
                    remarks = "(failed)";
                }
                
                //System.out.println("FG: "+formatOutput(fgradeTotal)+"% "+remarks);

            }
            else{
                System.out.println("\nInvalid Passing Grade Arguments!");
                System.exit(0);
            }
 
            return formatOutput(fgradeTotal)+"% "+remarks;
        }   
        
        public String formatOutput(float output){
            return new DecimalFormat("#,###.00").format(output);
        }
           
           
    }
